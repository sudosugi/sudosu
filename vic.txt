// victim.c - Safe test program for ptrace lab
#include <stdio.h>
#include <unistd.h>

int main(void) {
    pid_t pid = getpid();
    printf("Victim program running. PID = %d\n", pid);
    fflush(stdout);

    // Loop so the program stays alive
    while (1) {
        printf("Victim running...\n");
        fflush(stdout);
        sleep(2);
    }

    return 0;
}
